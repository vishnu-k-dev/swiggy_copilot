# 🧠 Step 1: Backend (Upgrade Version)

### `server.js`

```js
const express = require("express");
const foodItems = require("./data");
const app = express();

app.use(express.json());
app.use(express.static("public"));

let lastOrder = null;

function parseIntent(query) {
  query = query.toLowerCase();

  if (query.includes("same") || query.includes("again")) {
    return { repeat: true };
  }

  let budget = query.includes("cheap") ? 150 : null;
  let type = query.includes("veg") ? "veg" : null;

  return { budget, type };
}

function getResults(intent) {
  let results = foodItems;

  if (intent.type) {
    results = results.filter(f => f.type === intent.type);
  }

  if (intent.budget) {
    results = results.filter(f => f.price <= intent.budget);
  }

  return results.sort((a, b) => a.price - b.price).slice(0, 3);
}

app.post("/chat", (req, res) => {
  const { message } = req.body;
  const intent = parseIntent(message);

  if (intent.repeat && lastOrder) {
    return res.json({
      reply: `Reordering: ${lastOrder.name} - ₹${lastOrder.price}. Confirm?`
    });
  }

  const results = getResults(intent);

  if (!results.length) {
    return res.json({ reply: "No options found." });
  }

  lastOrder = results[0];

  const text = results.map((r, i) =>
    `${i + 1}. ${r.name} - ₹${r.price} - ${r.eta} mins`
  ).join("\n");

  res.json({
    reply: `🍽️ Top Picks (Powered by Swiggy)\n\n${text}\n\nReply 1/2/3`
  });
});

app.listen(3000, () => console.log("Running on http://localhost:3000"));
```

# 💬 Step 3: WhatsApp-style UI (🔥 This is the magic)

### `public/index.html`

```html
<!DOCTYPE html>
<html>
<head>
  <title>Swiggy Voice Copilot</title>
  <style>
    body {
      font-family: Arial;
      background: #111;
      color: white;
      display: flex;
      justify-content: center;
    }
    #chat {
      width: 400px;
      height: 600px;
      background: #1e1e1e;
      padding: 10px;
      overflow-y: auto;
      border-radius: 10px;
    }
    .msg {
      margin: 10px 0;
      padding: 10px;
      border-radius: 8px;
    }
    .user { background: #2563eb; text-align: right; }
    .bot { background: #374151; }
    input {
      width: 300px;
      padding: 10px;
    }
    button {
      padding: 10px;
    }
  </style>
</head>
<body>

<div>
  <h2>Swiggy Voice Copilot</h2>
  <div id="chat"></div>
  <input id="input" placeholder="Type or speak..." />
  <button onclick="send()">Send</button>
  <button onclick="startVoice()">🎤</button>
</div>

<script>
async function sendMessage(text) {
  addMessage(text, "user");

  const res = await fetch("/chat", {
    method: "POST",
    headers: {"Content-Type": "application/json"},
    body: JSON.stringify({ message: text })
  });

  const data = await res.json();
  addMessage(data.reply, "bot");
}

function addMessage(text, type) {
  const div = document.createElement("div");
  div.className = "msg " + type;
  div.innerText = text;
  document.getElementById("chat").appendChild(div);
}

function send() {
  const input = document.getElementById("input");
  sendMessage(input.value);
  input.value = "";
}

// 🎤 Voice (FREE browser API)
function startVoice() {
  const recognition = new (window.SpeechRecognition || window.webkitSpeechRecognition)();
  recognition.start();

  recognition.onresult = function(event) {
    const text = event.results[0][0].transcript;
    sendMessage(text);
  };
}
</script>

</body>
</html>
```
